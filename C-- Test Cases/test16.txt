int lol(int y, int u, int p){
    println(y);
    println(u);
    println(p);
}

int main() {
    int x,y;

    x = 123 + 456;
    println(x);

    y = 123 - 456;
    println(y);

    lol(x,y,-99);
    lol(y,x,99);

    x = 123 * 3;
    println(x);

    y = 123 / 3;
    println(x);

    lol(x,y,-88);
    lol(y,x,88);

    x = (123);
    println(x);

    y = -123;
    println(x);

    lol(x,y,-77);
    lol(y,x,77);
}
